  140.   /**
    141. 

  141.    * Combination of public events - connect, login, disconnect
    142. 

  142.    * @access public
    143. 

  143.    * @param string $host 
    144. 

  144.    * @param integer $port 
    145.

  149.   public function Authorise ($host, $port = false, $tval = false, $username, $password, $debug_level = 0) {
    150. 

  150.     $this->host = $host;
    151. 

  151.  
    152. 

  152.     //  If no port value is passed, retrieve it
    153. 

  153.     if ($port == false) {
    154. 

  154.       $this->port = $this->POP3_PORT;
    155. 

  155.     } else {
    156. 

  156.       $this->port = $port;
    157. 

  157.     }
    158. 

  158.  
    159. 

  159.     //  If no port value is passed, retrieve it
    160. 

  160.     if ($tval == false) {
    161. 

  161.       $this->tval = $this->POP3_TIMEOUT;
    162. 

  162.     } else {
    163. 

  163.       $this->tval = $tval;
    164. 

  164.     }
    165. 

  165.  
    166. 

  166.     $this->do_debug = $debug_level;
    167. 

  167.     $this->username = $username;
    168. 

  168.     $this->password = $password;
    169. 

  169.  
    170. 

  170.     //  Refresh the error log
    171. 

  171.     $this->error = null;
    172. 

  172.  
    173. 

  173.     //  Connect
    174. 

  174.     $result = $this->Connect($this->host, $this->port, $this->tval);
    175. 

  175.  
    176. 

  176.     if ($result) {
    177. 

  177.       $login_result = $this->Login($this->username, $this->password);
    178. 

  178.  
    179. 

  179.       if ($login_result) {
    180. 

  180.         $this->Disconnect();
    181. 

  181.  
    182. 

  182.         return true;
    183. 

  183.       }
    184. 

  184.  
    185. 

  185.     }
    186. 

  186.  
    187. 

  187.     //  We need to disconnect regardless if the login succeeded
    188. 

  188.     $this->Disconnect();
    189. 

  189.  
    190. 

  190.     return false;
    191. 

  191.   }

  193.   /**
    194. 

  194.    * Connect to the POP3 server
    195. 

  195.    * @access public
    196. 

  196.    * @param string $host 
    197. 

  197.    * @param integer $port 
    198. 

  198.    * @param integer $tval 
    199. 

  199.    * @return boolean 
    200. 

  200.    */
    201. 

  201.   public function Connect ($host, $port = false, $tval = 30) {
    202. 

  202.     //  Are we already connected?
    203. 

  203.     if ($this->connected) {
    204. 

  204.       return true;
    205. 

  205.     }
    206. 

  206.  
    207. 

  207.     /*
    208. 

  208.     On Windows this will raise a PHP Warning error if the hostname doesn't exist.
    209. 

  209.     Rather than supress it with @fsockopen, let's capture it cleanly instead
    210. 

  210.     */
    211. 

  211.  
    212. 

  212.     set_error_handler(array(&$this, 'catchWarning'));
    213. 

  213.  
    214. 

  214.     //  Connect to the POP3 server
    215. 

  215.     $this->pop_conn = fsockopen($host,    //  POP3 Host
    216. 

  216.                   $port,    //  Port #
    217. 

  217.                   $errno,   //  Error Number
    218. 

  218.                   $errstr,  //  Error Message
    219. 

  219.                   $tval);   //  Timeout (seconds)
    220. 

  220.  
    221. 

  221.     //  Restore the error handler
    222. 

  222.     restore_error_handler();
    223. 

  223.  
    224. 

  224.     //  Does the Error Log now contain anything?
    225. 

  225.     if ($this->error && $this->do_debug >= 1) {
    226. 

  226.       $this->displayErrors();
    227. 

  227.     }
    228. 

  228.  
    229. 

  229.     //  Did we connect?
    230. 

  230.     if ($this->pop_conn == false) {
    231. 

  231.       //  It would appear not...
    232. 

  232.       $this->error = array(
    233. 

  233.         'error' => "Failed to connect to server $host on port $port",
    234. 

  234.         'errno' => $errno,
    235. 

  235.         'errstr' => $errstr
    236. 

  236.       );
    237. 

  237.  
    238. 

  238.       if ($this->do_debug >= 1) {
    239. 

  239.         $this->displayErrors();
    240. 

  240.       }
    241. 

  241.  
    242. 

  242.       return false;
    243. 

  243.     }
    244. 

  244.  
    245. 

  245.     //  Increase the stream time-out
    246. 

  246.  
    247. 

  247.     //  Check for PHP 4.3.0 or later
    248. 

  248.     if (version_compare(phpversion(), '5.0.0', 'ge')) {
    249. 

  249.       stream_set_timeout($this->pop_conn, $tval, 0);
    250. 

  250.     } else {
    251. 

  251.       //  Does not work on Windows
    252. 

  252.       if (substr(PHP_OS, 0, 3) !== 'WIN') {
    253. 

  253.         socket_set_timeout($this->pop_conn, $tval, 0);
    254. 

  254.       }
    255. 

  255.     }
    256. 

  256.  
    257. 

  257.     //  Get the POP3 server response
    258. 

  258.     $pop3_response = $this->getResponse();
    259. 

  259.  
    260. 

  260.     //  Check for the +OK
    261. 

  261.     if ($this->checkResponse($pop3_response)) {
    262. 

  262.     //  The connection is established and the POP3 server is talking
    263. 

  263.     $this->connected = true;
    264. 

  264.       return true;
    265. 

  265.     }
    266. 

  266.  
    267. 

  267.   }

  269.   /**
    270. 

  270.    * Login to the POP3 server (does not support APOP yet)
    271. 

  271.    * @access public
    272. 

  272.    * @param string $username 
    273. 

  273.    * @param string $password 
    274. 

  274.    * @return boolean 
    275. 

  275.    */
    276. 

  276.   public function Login ($username = '', $password = '') {
    277. 

  277.     if ($this->connected == false) {
    278. 

  278.       $this->error = 'Not connected to POP3 server';
    279. 

  279.  
    280. 

  280.       if ($this->do_debug >= 1) {
    281. 

  281.         $this->displayErrors();
    282. 

  282.       }
    283. 

  283.     }
    284. 

  284.  
    285. 

  285.     if (empty($username)) {
    286. 

  286.       $username = $this->username;
    287. 

  287.     }
    288. 

  288.  
    289. 

  289.     if (empty($password)) {
    290. 

  290.       $password = $this->password;
    291. 

  291.     }
    292. 

  292.  
    293. 

  293.     $pop_username = "USER $username" . $this->CRLF;
    294. 

  294.     $pop_password = "PASS $password" . $this->CRLF;
    295. 

  295.  
    296. 

  296.     //  Send the Username
    297. 

  297.     $this->sendString($pop_username);
    298. 

  298.     $pop3_response = $this->getResponse();
    299. 

  299.  
    300. 

  300.     if ($this->checkResponse($pop3_response)) {
    301. 

  301.       //  Send the Password
    302. 

  302.       $this->sendString($pop_password);
    303. 

  303.       $pop3_response = $this->getResponse();
    304. 

  304.  
    305. 

  305.       if ($this->checkResponse($pop3_response)) {
    306. 

  306.         return true;
    307. 

  307.       } else {
    308. 

  308.         return false;
    309. 

  309.       }
    310. 

  310.     } else {
    311. 

  311.       return false;
    312. 

  312.     }
    313. 

  313.   }

    320.

    321. 

  321.  
    322. 

  322.     fclose($this->pop_conn);
    323.

  356.   /**
    357. 

  357.    * Checks the POP3 server response for +OK or -ERR

  359.    * @param string $string 
    360. 

  360.    * @return boolean 
    361. 

  361.    */

  362.   private function checkResponse ($string) {
    363. 

  363.     if (substr($string, 0, 3) !== '+OK') {
    364. 

  364.       $this->error = array(
    365. 

  365.         'error' => "Server reported an error: $string",
    366. 

  366.         'errno' => 0,
    367. 

  367.         'errstr' => ''
    368. 

  368.       );
    369. 

  369.  
    370. 

  370.       if ($this->do_debug >= 1) {
    371. 

  371.         $this->displayErrors();
    372. 

  372.       }
    373. 

  373.  
    374. 

  374.       return false;
    375. 

  375.     } else {
    376. 

  376.       return true;
    377. 

  377.     }
    378. 

  378.  
    379. 

  379.   }

    386. 

  386.     echo '<pre>';
    387. 

  387.  
    388. 

  388.     foreach ($this->error as $single_error) {
    389. 

  389.       print_r($single_error);
    390. 

  390.     }
    391. 

  391.  
    392. 

  392.     echo '</pre>';
    393. 

  393.   }

  403.   private function catchWarning ($errno, $errstr, $errfile, $errline) {
    404. 

  404.     $this->error[] = array(
    405. 

  405.       'error' => "Connecting to the POP3 server raised a PHP warning: ",
    406. 

  406.       'errno' => $errno,
    407. 

  407.       'errstr' => $errstr
    408. 

  408.     );
