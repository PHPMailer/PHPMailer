    407. 

  407.     $address = trim($address);
    408. 

  408.     if (!preg_match('/^(to|cc|bcc)$/', $kind)) {
    409. 

  409.       echo 'Invalid recipient array: ' . kind;
    410. 

  410.       return false;
    411. 

  411.     }
    412. 

  412.     if (!self::ValidateAddress($address)) {
    413. 

  413.       if ($this->exceptions) {
    414. 

  414.         throw new phpmailerException('Invalid address: '.$address);
    415. 

  415.       }
    416. 

  416.       echo 'Invalid address: '.$address;
    417. 

  417.       return false;
    418. 

  418.     }
    419. 

  419.     if (!isset($this->all_recipients[strtolower($address)])) {
    420. 

  420.       array_push($this->$kind, array($address, $name));
    421. 

  421.       $this->all_recipients[strtolower($address)] = true;
    422. 

  422.       return true;
    423. 

  423.     }
    424. 

  424.     return false;
    425.

    431. 

  431.    * Conforms approximately to RFC2822

  432.    * @link http://www.hexillion.com/samples/#Regex Original pattern found here
    433. 

  433.    * @param string $address The email address to check

  438.   public static function ValidateAddress($address) {
    439. 

  439.     if (function_exists('filter_var')) { //Introduced in PHP 5.2
    440. 

  440.       if(filter_var($address, FILTER_VALIDATE_EMAIL) === FALSE) {
    441. 

  441.         return false;
    442. 

  442.       } else {
    443. 

  443.         return true;
    444. 

  444.       }
    445. 

  445.     } else {
    446. 

  446.       return preg_match('/^(?:[\w\!\#\$\%\&\'\*\+\-\/\=\?\^\`\{\|\}\~]+\.)*[\w\!\#\$\%\&\'\*\+\-\/\=\?\^\`\{\|\}\~]+@(?:(?:(?:[a-zA-Z0-9_](?:[a-zA-Z0-9_\-](?!\.)){0,61}[a-zA-Z0-9_-]?\.)+[a-zA-Z0-9_](?:[a-zA-Z0-9_\-](?!$)){0,61}[a-zA-Z0-9_]?)|(?:\[(?:(?:[01]?\d{1,2}|2[0-4]\d|25[0-5])\.){3}(?:[01]?\d{1,2}|2[0-4]\d|25[0-5])\]))$/', $address);
    447. 

  447.     }
    448. 

  448.   }
